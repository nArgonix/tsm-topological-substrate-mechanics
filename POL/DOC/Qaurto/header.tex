% DYNAMICZNY ZNACZNIK: automatycznie zamienia "Rozdział" na "Dodatek"
\renewcommand*{\chapterformat}{\chapapp~\thechapter:}

% Automatyczne wykrywanie sekcji załączników (Appendix)
\newcommand*{\isappendix}{0}
\makeatletter
\g@addto@macro\appendix{\gdef\isappendix{1}}
\makeatother

% Główny mechanizm sterujący wyglądem nagłówków rozdziałów i dodatków
\renewcommand*{\chapterlineswithprefixformat}[3]{%
  \ifnum\isappendix=1
    % =========================================================================
    % DODATEK (APPENDIX) - PREFIKS PO LEWEJ, TYTUŁ NA ŚRODKU
    % =========================================================================
    {\raggedright
      {\large\bfseries #2}\par       % "Dodatek A:" na górze po lewej
    }%
    \nopagebreak\vspace*{0.6\baselineskip}     % Odstęp pomiędzy liniami
    {\centering
      {\Huge\bfseries #3}\par        % Sam tytuł dodatku na środku
    }%
  \else
    % =========================================================================
    % ZWYKŁE ROZDZIAŁY - PREFIKS PO LEWEJ, TYTUŁ NA ŚRODKU
    % =========================================================================
    {\raggedright
      {\large\bfseries #2}\par       % "Rozdział X:" na górze po lewej
    }%
    \nopagebreak\vspace*{0.6\baselineskip}     % Odstęp między prefiksem a tytułem
    {\centering
      {\Huge\bfseries #3}\par        % Sam tytuł rozdziału na środku
    }%
  \fi
}

% Format dla Spisu Treści i Wstępu - NA ŚRODKU
\renewcommand*{\chapterlinesformat}[3]{%
  {\centering
    {\Huge\bfseries #3}\par          % Tytuł nienumerowany na środku
  }%
}
